 17 var Webview,
 18     utils = require("./utils"),
 19     webkitEvent = require('./webkitEvent'),
 20     qnx = utils.getQnxNamespace(),
 21     _self;
 22 
 23 /**
 24  * @class A javascript abstraction of the underlying webkit webview
 25  * @property {Number} id The id of the webview. READ ONLY
 26  * @property {Number} windowGroup The windowGroup of the webview. READ ONLY
 27  * @property {Boolean} visible Whether the webview is currently visible
 28  * @property {Boolean} active Whether the webview is currently active
 29  * @property {Number} zOrder The zOrder of the webview
 30  */
 31 function Webview() {
 32 
 33     var chromeId = 1,
 34         windowGroup = qnx.callExtensionMethod("webview.applicationWindowGroup", chromeId),
 35         id = parseInt(qnx.callExtensionMethod('webview.create', windowGroup, "InProcess"), 10),
 36         visible = false,
 37         active = false,
 38         zOrder = -99;
 39 
 40     this.__defineGetter__("id", function () {
 41         return id;
 42     });
 43 
 44     this.__defineGetter__("windowGroup", function () {
 45         return windowGroup;
 46     });
 47 
 48     this.__defineGetter__("visible", function () {
 49         return visible;
 50     });
 51 
 52     this.__defineSetter__("visible", function (newVisibility) {
 53         visible = !!newVisibility;
 54         qnx.callExtensionMethod("webview.setVisible", id, visible);
 55     });
 56 
 57     this.__defineGetter__("active", function () {
 58         return active;
 59     });
 60 
 61     this.__defineSetter__("active", function (newActivity) {
 62         active = !!newActivity;
 63         qnx.callExtensionMethod("webview.setActive", id, active);
 64     });
 65 
 66     this.__defineGetter__("zOrder", function () {
 67         return zOrder;
 68     });
 69 
 70     this.__defineSetter__("zOrder", function (newZOrder) {
 71         zOrder = parseInt(newZOrder, 10);
 72         qnx.callExtensionMethod("webview.setZOrder", id, zOrder);
 73     });
 74     
 75 }
 76 
 77 Webview.prototype.on = function (eventType, eventListener) {
 78     webkitEvent.on({id: this.id, eventType: eventType}, eventListener); //What to put for scope???
 79 };
 80 Webview.prototype.once  = function (eventType, eventListener) {
 81     webkitEvent.once({id: this.id, eventType: eventType}, eventListener); //What to put for scope???
 82 };
 83 Webview.prototype.clear = function (eventType) {
 84     webkitEvent.clear({id: this.id, eventType: eventType});
 85 };

 88 /**
 89 * @description Sets the location of the webview on the screen
 90 * @param {Number} x The x position of the webview
 91 * @param {Number} y The y position of the webview
 92 * @param {Number} width The width of the webview
 93 * @param {Number} height The height of the webview
 94 */
 95 Webview.prototype.setGeometry = function (x, y, width, height) {
 96     qnx.callExtensionMethod("webview.setGeometry", this.id, x, y, width, height);
 97 };
 98 
 99 /**
100 * @description Destroys the underlying native webview
101 * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
102 */ 
103 Webview.prototype.destroy = function (onComplete) {
104     qnx.callExtensionMethod("webview.destroy", this.id);
105 
106     if (onComplete && typeof onComplete === 'function') {
107         onComplete();
108     }
109 };
110 
111 /**
112 * @description Sets the URL on the webview
113 * @param {String} url The fully qualified url to be loaded into the webview
114 */ 
115 Webview.prototype.setURL = function (url) {
116     qnx.callExtensionMethod("webview.loadURL", this.id, url);
117 };
118 
119 /**
120 * @description Executes javascript in the context of the webview
121 * @param {String} js The javascript expression to be executed
122      * @param {boolean} [inIsolatedWorld=false] Run javascript in isolated context
123      * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
124 */ 
125 Webview.prototype.executeJavaScript = function (js, inIsolatedWorld, onComplete) {
126     qnx.callExtensionMethod("webview.executeJavaScript", this.id, js, inIsolatedWorld || false);
127 
128     if (onComplete && typeof onComplete === 'function') {
129         onComplete();
130     }
131 };
132 
133 /**
134      * @description Sets the background color of the webview
135      * @param {String} color The desired background color of the webview in hex
136      * @example webview.setBackgroundColor("0x00FFFFFF");
137 */
138 Webview.prototype.setBackgroundColor = function (color) {
139     qnx.callExtensionMethod("webview.setBackgroundColor", this.id, color);
140 };
